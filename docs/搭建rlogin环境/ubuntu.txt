# 1. 安装依赖
apt install -y rsh-server
# 可选，如果没有就安装
apt install -y xinetd
# 2. 启用配置
touch /etc/xinetd.d/rlogin
vim /etc/xinetd.d/rlogin
service login
{
    socket_type     = stream
    protocol        = tcp
    wait            = no
    user            = root
    server          = /usr/sbin/in.rlogind
    disable         = no
}
# 3. 设置可访问ip
# 这个可选
vim /etc/hosts.equiv
# 可访问的ip列表
192.168.3.11
# 建议这个
vim /etc/hosts.allow
# 设置可访问网段
rlogind: 192.168.3.0/24
# 4. 关闭防火墙(或者自己添加端口)
systemctl stop ufw
systemctl disable ufw
# 5. 启动并启用xinetd服务
# 启动
systemctl start xinetd
# 重启
systemctl restart xinetd
# 启用
systemctl enable xinetd
# 状态
systemctl status xinetd
