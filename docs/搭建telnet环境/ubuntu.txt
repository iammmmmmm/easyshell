# 1. 安装依赖
apt install -y telnetd xinetd
# 2. 启用配置
touch /etc/xinetd.d/telnet
vim /etc/xinetd.d/telnet
service telnet
{
    disable = no
    socket_type = stream
    protocol = tcp
    wait = no
    user = root
    server = /usr/sbin/in.telnetd
    log_on_failure += USERID
}
# 3. 链接文件
ln -s /usr/sbin/telnetd /usr/sbin/in.telnetd
chmod 755 /usr/sbin/in.telnetd
# 4. 关闭防火墙(或者自己添加端口)
systemctl stop ufw
systemctl disable ufw
# 5. 启动并启用telnet服务
systemctl start xinetd
systemctl enable xinetd
